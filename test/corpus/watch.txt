===
Watch with http
===

job web {
  run "web-server --port 8080"

  watch health {
    http "http://localhost:8080/health" {
      status = 200
    }
    initial_delay = 5s
    poll = 10s
    threshold = 3
    on_fail shutdown
  }
}

===
Watch with spawn
===

job web {
  run "web-server"

  watch disk {
    exists "/var/run/healthy"
    on_fail spawn @recovery
  }
}
